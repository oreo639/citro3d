740 /**
741  * @brief Quaternion inverse
742  * @note This is equivalent to `Quat_Pow(v, -1)`
743  * @param[in] q Quaternion of which to get inverse
744  * @return q<sup>-1</sup>
745  */
746 static inline C3D_FQuat Quat_Inverse(C3D_FQuat q)
747 {
748  // q^-1 = (q.r - q.i - q.j - q.k) / (q.r^2 + q.i^2 + q.j^2 + q.k^2)
749  // = q* / (q∙q)
750  C3D_FQuat c = Quat_Conjugate(q);
751  float d = Quat_Dot(q, q);
752  return Quat_New(c.i/d, c.j/d, c.k/d, c.r/d);
753 }
